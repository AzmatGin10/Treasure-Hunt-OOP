        while bfs_queue:
            (current_x, current_y), current_distance = bfs_queue.pop(0)

            if current_distance > self.min_distance:

                if 0 < new_x < self.mazeSize and 0 < new_y < self.mazeSize and not self.maze[new_y][new_x]:
